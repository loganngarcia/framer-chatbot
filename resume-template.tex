% Custom commands

\newcommand{\resumeItem}[1]{
  \item\small{
    #1 \vspace{-2pt}
  }
}

\newcommand{\resumeItemWithoutTitle}[1]{
  \item{
    #1 \vspace{-2pt}
  }
}

% Bulletless subheading block:
% Left row 1 = bold title line
% Right row 1 = date
% Left row 2 = secondary line (can be empty)
% Right row 2 = location in 80% gray, non-italic
\newcommand{\resumeSubheading}[4]{
  \vspace{-1pt}
  \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
    \textbf{#1} & #4 \\
    #3 & \textcolor[gray]{0.80}{#2} \\
  \end{tabular*}\vspace{-5pt}
}

% Backwards-compatible helper (kept, but no bolding)
\newcommand{\resumeSubItem}[2]{\resumeItemWithoutTitle{#1: #2}\vspace{-3pt}}

\renewcommand{\labelitemii}{$\bullet$}

% Keep list wrappers but make them bulletless and consistent
\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=0pt,label={},itemsep=0pt,topsep=0pt]}
\newcommand{\resumeSubHeadingListEnd}{\end{itemize}}

% Description bullets only (using filled bullets)
\newcommand{\resumeItemListStart}{\vspace{-12pt}\begin{itemize}[leftmargin=1.2em,itemsep=2pt,topsep=0pt,label=$\bullet$]}
\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-5pt}}
